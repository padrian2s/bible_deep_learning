import numpy as np import matplotlib.pyplot as plt def gabor_bank(size=32, n_orientations=8, n_frequencies=4): """Genereaza o banca de filtre Gabor ca in V1""" filters = [] for f_idx in range(n_frequencies): frequency = 0.1 + f_idx * 0.15 # Frecvente crescatoare for o_idx in range(n_orientations): orientation = o_idx * np.pi / n_orientations # Genereaza filtru Gabor x = np.linspace(-1, 1, size) y = np.linspace(-1, 1, size) X, Y = np.meshgrid(x, y) X_rot = X * np.cos(orientation) + Y * np.sin(orientation) Y_rot = -X * np.sin(orientation) + Y * np.cos(orientation) gaussian = np.exp(-0.5 * (X_rot**2 + Y_rot**2) / 0.3**2) sinusoid = np.cos(2 * np.pi * frequency * X_rot * size) filters.append(gaussian * sinusoid) return np.array(filters) # Total: 8 orientari Ã— 4 frecvente = 32 filtre # Similar cu primul layer al unui CNN antrenat pe imagini!

Una dintre cele mai izbitoare corespondente intre neurostiinta si ML: Olshausen si Field (1996) au aratat ca un simplu algoritm nesupervizat (sparse coding) aplicat pe patch-uri de imagini naturale invata features similare cu celulele simple din V1! De atunci, MULTE algoritmi de invatare (inclusiv primul layer al CNN-urilor) produc filtre Gabor-like.
